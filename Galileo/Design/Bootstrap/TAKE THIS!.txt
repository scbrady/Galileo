		                var caption = item.user_id;

                            '<span class="selectize-caption">' + escape(caption) + '</span>' +
                        '</div>';
		            }
		        },
		        load: function (query, callback) {
		            if (!query.length) return callback();

		                url: '/Comment/Users',
		                dataType: 'json',
		                success: function (data) {
		                    callback(data);
		                }
		            });
		        }
